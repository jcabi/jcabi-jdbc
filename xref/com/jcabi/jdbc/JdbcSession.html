47  /**
48   * Universal JDBC wrapper.
49   *
50   * <p>Execute a simple SQL query over a JDBC data source:
51   *
52   * <pre> String name = new JdbcSession(source)
53   *   .sql("SELECT name FROM foo WHERE id = ?")
54   *   .set(123)
55   *   .select(
56   *     new JdbcSession.Handler&lt;String&gt;() {
57   *       &#64;Override
58   *       public String handle(final ResultSet rset) throws SQLException {
59   *         rset.next();
60   *         return rset.getString(1);
61   *       }
62   *     }
63   *   );</pre>
64   *
65   * <p>There are a number of convenient pre-defined handlers, like
66   * {@link VoidHandler}, {@link NotEmptyHandler}, {@link SingleHandler}, etc.
67   *
68   * <p>Methods {@link #insert(JdbcSession.Handler)},
69   * {@link #update(JdbcSession.Handler)},
70   * {@link #execute()}, and
71   * {@link #select(JdbcSession.Handler)} clean the list of arguments pre-set by
72   * {@link #set(Object)}. The class can be used for a complex transaction, when
73   * it's necessary to perform a number of SQL statements in a group. For
74   * example, the following construct will execute two SQL queries, in a single
75   * transaction and will "commit" at the end (or rollback the entire transaction
76   * in case of any error in between):
77   *
78   * <pre> new JdbcSession(source)
79   *   .autocommit(false)
80   *   .sql("START TRANSACTION")
81   *   .execute()
82   *   .sql("DELETE FROM foo WHERE id = ?")
83   *   .set(444)
84   *   .execute()
85   *   .set(555)
86   *   .execute()
87   *   .commit();</pre>
88   *
89   * <p>The following SQL queries will be sent to the database:
90   *
91   * <pre> START TRANSACTION;
92   * DELETE FROM foo WHERE id = 444;
93   * DELETE FROM foo WHERE id = 555;
94   * COMMIT;</pre>
95   *
96   * <p>{@link #autocommit(boolean)} (with {@code false} as an argument)
97   * can be used when it's necessary to execute
98   * a statement and leave the connection open. For example when shutting down
99   * the database through SQL:
100  *
101  * <pre> new JdbcSession(&#47;* H2 Database data source *&#47;)
102  *   .autocommit(false)
103  *   .sql("SHUTDOWN COMPACT")
104  *   .execute();</pre>
105  *
106  * <p>This class is thread-safe.
107  *
108  * @author Yegor Bugayenko (yegor@tpc2.com)
109  * @version $Id$
110  * @since 0.1.8
111  */

238     /**
239      * Make SQL {@code INSERT} request.
240      *
241      * <p>{@link JdbcSession.Handler} will receive
242      * a {@link ResultSet} of generated keys.
243      *
244      * <p>JDBC connection is opened and, optionally, closed by this method.
245      *
246      * @param handler The handler or result
247      * @return The result
248      * @param <T> Type of response
249      * @throws SQLException If fails
250      */
251     public <T> T insert(
252         @NotNull(message = "handler can't be NULL")
253         final JdbcSession.Handler<T> handler)
254         throws SQLException {
255         return this.run(
256             handler,
257             new JdbcSession.Fetcher() {
258                 @Override
259                 public ResultSet fetch(final PreparedStatement stmt)
260                     throws SQLException {
261                     stmt.execute();
262                     return stmt.getGeneratedKeys();
263                 }
264                 @Override
265                 public PreparedStatement statement(final Connection conn)
266                     throws SQLException {
267                     return conn.prepareStatement(
268                         JdbcSession.this.query,
269                         Statement.RETURN_GENERATED_KEYS
270                     );
271                 }
272             }
273         );
274     }

378     /**
379      * Run this handler, and this fetcher.
380      * @param handler The handler or result
381      * @param fetcher Fetcher of result set
382      * @return The result
383      * @param <T> Type of response
384      * @throws SQLException If fails
385      * @checkstyle ExecutableStatementCount (100 lines)
386      */
387     private <T> T run(final JdbcSession.Handler<T> handler,
388         final JdbcSession.Fetcher fetcher)
389         throws SQLException {
390         if (this.query == null) {
391             throw new IllegalStateException("call #sql() first");
392         }
393         final Connection conn = this.connect();
394         T result;
395         try {
396             conn.setAutoCommit(this.auto);
397             final PreparedStatement stmt = fetcher.statement(conn);
398             try {
399                 this.parametrize(stmt);
400                 final ResultSet rset = fetcher.fetch(stmt);
401                 // @checkstyle NestedTryDepth (5 lines)
402                 try {
403                     result = handler.handle(rset, stmt);
404                 } finally {
405                     if (rset != null) {
406                         rset.close();
407                     }
408                 }
409             } finally {
410                 stmt.close();
411             }
412         } catch (SQLException ex) {
413             if (!this.auto) {
414                 conn.rollback();
415                 this.disconnect();
416             }
417             throw new SQLException(ex);
418         } finally {
419             if (this.auto) {
420                 this.disconnect();
421             }
422             this.args.clear();
423         }
424         return result;
425     }

427     /**
428      * Open connection and cache it locally in the class.
429      * @return Connection to use
430      * @throws SQLException If fails
431      */
432     private Connection connect() throws SQLException {
433         synchronized (this.args) {
434             if (this.connection.get() == null) {
435                 this.connection.set(this.source.getConnection());
436             }
437             return this.connection.get();
438         }
439     }

441     /**
442      * Close connection if it's open (runtime exception otherwise).
443      * @throws SQLException If fails to do the SQL operation
444      */
445     private void disconnect() throws SQLException {
446         final Connection conn = this.connection.getAndSet(null);
447         if (conn == null) {
448             throw new IllegalStateException(
449                 "connection is not open, can't close"
450             );
451         }
452         conn.close();
453     }

455     /**
456      * Add params to the statement.
457      * @param stmt The statement to parametrize
458      * @throws SQLException If some problem
459      */
460     private void parametrize(final PreparedStatement stmt) throws SQLException {
461         int pos = 1;
462         for (final Object arg : this.args) {
463             if (arg == null) {
464                 stmt.setString(pos, null);
465             } else if (arg instanceof Long) {
466                 stmt.setLong(pos, Long.class.cast(arg));
467             } else if (arg instanceof Boolean) {
468                 stmt.setBoolean(pos, Boolean.class.cast(arg));
469             } else if (arg instanceof Date) {
470                 stmt.setDate(pos, Date.class.cast(arg));
471             } else if (arg instanceof Integer) {
472                 stmt.setInt(pos, Integer.class.cast(arg));
473             } else if (arg instanceof Utc) {
474                 Utc.class.cast(arg).setTimestamp(stmt, pos);
475             } else {
476                 stmt.setString(pos, arg.toString());
477             }
478             ++pos;
479         }

486     public interface Handler<T> {
